# 1. Build Stage
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy the entire repository into the container
COPY . .

# Move into the subfolder where the .csproj file lives
WORKDIR "/src/MyWebApp"

# Restore dependencies
RUN dotnet restore

# Build and publish the app to a folder called /app
RUN dotnet publish -c Release -o /app

# 2. Run Stage
FROM mcr.microsoft.com/dotnet/aspnet:8.0
WORKDIR /app

# Install libgdiplus for the BOOSE graphics engine
RUN apt-get update && apt-get install -y libgdiplus libc6-dev && rm -rf /var/lib/apt/lists/*

# Copy the compiled files from the build stage
COPY --from=build /app .

# Expose the port ASP.NET uses by default in .NET 8+
ENV ASPNETCORE_URLS=http://+:8080
EXPOSE 8080

# Start the application
ENTRYPOINT ["dotnet", "MyWebApp.dll"]
